145   /**
146    * get the name space name
147    * 
148    * @param pageTitle
149    * @return
150    */
151   public static String getNameSpaceName(String pageTitle) {
152     Matcher matcher = namespacePattern.matcher(pageTitle);
153     if (matcher.find()) {
154       // LOGGER.log(Level.INFO,pageTitle);
155       return matcher.group(1);
156     }
157     return null;
158   }
159 
160   /**
161    * construct me from a localized pageTitle in a given language
162    * 
163    * @param pageTitle
164    * @param siteinfo
165    * @throws Exception
166    */
167   public PageInfo(String pageTitle, SiteInfo siteinfo) throws Exception {
168     this.pageTitle = pageTitle;
169     this.canonicalPageTitle = pageTitle;
170     this.siteinfo = siteinfo;
171     nameSpaceName = getNameSpaceName(pageTitle);
172     this.lang = siteinfo.getLang();
173     namespace = null;
174     // do we have to map a namespace?
175     if (nameSpaceName != null) {
176       // get the namespace
177       namespace = siteinfo.getNamespaces().get(nameSpaceName);
178       if (namespace == null) {
179         String lookupName=nameSpaceName;
180         // get it from the canonical namespaces
181         namespace = siteinfo.getNamespacesByCanonicalName().get(lookupName);
182       }
183       // ok we found one
184       if (namespace != null) {
185         String canonical = namespace.getCanonical();
186         this.canonicalPageTitle = pageTitle.replaceFirst(nameSpaceName + ":",
187             canonical + ":");
188         namespaceId = namespace.getId();
189       } else {
190         if (debug) {
191           LOGGER.log(Level.WARNING, "namespace for namespacename '"
192               + nameSpaceName + "' not found");
193           for (Ns ns : siteinfo.getNamespacesByCanonicalName().values()) {
194             LOGGER.log(Level.WARNING, ns.getValue() + "->" + ns.getCanonical());
195           }
196         }
197       }
198     }
199     // normalize the title
200     this.canonicalPageTitle = this.canonicalPageTitle.replace(" ", "_"); 
201   }

207   public String dual() {

209      return result;
