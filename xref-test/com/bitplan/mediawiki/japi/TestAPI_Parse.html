41    @Test
42    public void testGetSectionList() throws Exception {
43      // http://stackoverflow.com/questions/16840447/retrieve-the-content-of-a-section-via-mediawiki-api
44      Mediawiki wiki = new Mediawiki("https://en.wikipedia.org", "/w");
45      // wiki.debug=true;
46      String[] pageTitles = { "License", "Hierarchy" };
47      // debug=true;
48      for (String pageTitle : pageTitles) {
49        List<S> sections = wiki.getSections(pageTitle);
50        assertNotNull(sections);
51        assertTrue(sections.size() > 5);
52        for (S section : sections) {
53          String sectionMsg = section.getIndex() + "-" + section.getLevel() + ":"
54              + section.getToclevel() + " " + section.getNumber() + " "
55              + section.getAnchor();
56          if (debug) {
57            LOGGER.log(Level.INFO, sectionMsg);
58          }
59        }
60      }
61    }
